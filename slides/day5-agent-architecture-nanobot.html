Agent 循环 (Agent Loop)

while not done:
    # 1. 观察：收集信息
    observation = get_user_input() + get_memory()
    
    # 2. 思考：LLM 决策
    thought = llm.think(observation)
    
    # 3. 行动：执行工具或回复
    if thought.needs_tool:
        result = execute_tool(thought.tool, thought.args)
        observation += result
    else:
        respond(thought.answer)
        done = True

Agent 核心循环 (简化版)

# agent/core.py 核心逻辑
class Agent:
    def __init__(self, config):
        self.llm = get_provider(config)
        self.memory = Memory()
        self.tools = load_tools()
    
    async def run(self, message: str) -> str:
        # 构建提示词
        prompt = self.build_prompt(message)
        
        # 调用 LLM
        response = await self.llm.chat(prompt, tools=self.tools)
        
        # 处理工具调用
        while response.tool_calls:
            for call in response.tool_calls:
                result = await self.execute_tool(call)
                prompt.append({"role": "tool", "content": result})
            response = await self.llm.chat(prompt, tools=self.tools)
        
        # 保存记忆
        self.memory.save(message, response.content)
        
        return response.content

记忆系统实现

# agent/memory.py
class Memory:
    def __init__(self, path="~/.nanobot/memory"):
        self.path = Path(path).expanduser()
        self.short_term = []  # 当前会话
        self.long_term = self.load_long_term()  # 持久化
    
    def save(self, user_msg, bot_msg):
        # 短期记忆：追加到对话历史
        self.short_term.append({"role": "user", "content": user_msg})
        self.short_term.append({"role": "assistant", "content": bot_msg})
        
        # 长期记忆：定期总结并存储
        if len(self.short_term) > 20:
            summary = self.summarize(self.short_term)
            self.long_term.append(summary)
            self.persist()
    
    def get_context(self, limit=10):
        return self.short_term[-limit:] + self.long_term[-5:]

装饰器模式，简单直观

🚀 2分钟部署 Agent

Step 1: 安装

# 方式一：pip
pip install nanobot-ai

# 方式二：uv (更快)
uv tool install nanobot-ai

# 方式三：源码 (开发用)
git clone https://github.com/HKUDS/nanobot.git
cd nanobot && pip install -e .

Step 2: 配置

# 初始化配置
nanobot onboard
// ~/.nanobot/config.json
{
  "providers": {
    "openrouter": {
      "apiKey": "sk-or-v1-xxx"
    }
  },
  "agents": {
    "defaults": {
      "model": "anthropic/claude-sonnet-4-5"
    }
  }
}

API Key: openrouter.ai/keys

Step 3: 运行

# 命令行模式
nanobot agent -m "What is 2+2?"

# 或启动 Gateway (连接聊天平台)
nanobot gateway

✅ 完成！你现在有了一个可工作的 AI Agent
